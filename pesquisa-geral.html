Passo 1:

Para realizar a busca de dados no banco de dados, é necessário a criação de um arquivo .php que tenha acesso aos dados do banco de dados e exiba para o usuário. Seguindo o exemplo do nome e do email, temos:



<?php
include_once('conexao.php');
//executa a query com base na conexão
$query=mysqli_query($conexao, "select * from cadastro");
//verifica se a query esta ok
if(!$query)
{
die('Query Inválida' . @mysqli_error($conexao)); // mostra o erro
}
//retorna uma matriz que corresponde a linha - ponteiro
//$dados = mysqli_fetch_array ($query); -------> mostra só o 1º dado
//recupera as informações do array $dados, usando o nome do campo como referência
//listando todos os arquivos da tabela
echo "<center>";
echo "<table border='1px'>";
echo "<tr><th> Matrícula </th><th> Nome </th><th>Email</th></tr>";
while ($dados = mysqli_fetch_array($query))
{
//listando todos os arquivos da tabela
echo "<tr>";
echo "<td>".$dados['nome']."</td>";
echo "<td>".$dados['email']."</td>";
}
echo"</table>";
echo "";
//finaliza a conexao
mysqli_close($conexao);
?>
